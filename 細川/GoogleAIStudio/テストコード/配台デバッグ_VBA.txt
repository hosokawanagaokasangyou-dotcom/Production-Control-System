' =========================================================
' 配台デバッグ: 1 ロールごとのトレースを Excel で確認する（標準モジュールに貼り付け）
' =========================================================
' 前提: xlwings アドイン有効・Show Console（推奨）
'
' 【呼び出し】マクロブックと同じフォルダに xlwings.conf.json（PYTHONPATH=python）を置く。
'   VBA では「ツール → 参照」で xlwings を有効にし、次のように **RunPython だけ** を呼ぶ
'   （xlwings.RunPython のような修飾は使わない。存在しない／解決できず止まる原因になる）。
'   例:
'     RunPython "import xlwings_console_runner as _xw; _xw.run_stage1_for_xlwings()"
'     RunPython "import xlwings_console_runner as _xw; _xw.run_refresh_dispatch_roll_trace_sheet_for_xlwings()"
'
' 【仕組み】
'   planning_core が各ロール割付成功時に log/dispatch_roll_trace.jsonl へ1行ずつ追記する。
'   Python が「配台_ロールトレース」シートに一覧を流し込む。
'   人が行を順に見て「どの順で・どのタスクが・どの設備に入ったか」を現認する。
'
' 環境変数（任意・Windows のシステム環境 or バッチ起動で設定）:
'   DISPATCH_ROLL_TRACE_JSONL=log/dispatch_roll_trace.jsonl  … JSONL のパス
'   DISPATCH_DEBUG_STOP_AFTER_ROLLS=5  … 5ロールで段階2の割付シミュレーションを打ち切り（部分結果で原因調査）
'
' RunPython 文字列は1行（改行不可）のため、下記は1行にまとめて登録してください。
' =========================================================

' xlwings 参照設定後のグローバル RunPython（python フォルダは xlwings.conf.json の PYTHONPATH）
Private Sub DbgRunConsoleRunner(ByVal entryPoint As String)
    On Error GoTo EH
    RunPython "import xlwings_console_runner as _xw; _xw." & entryPoint & "()"
    Exit Sub
EH:
    Err.Raise Err.Number, "DbgRunConsoleRunner", "RunPython: " & Err.Description
End Sub

Private Function DbgRollTraceJsonlPath() As String
    DbgRollTraceJsonlPath = ThisWorkbook.Path & "\log\dispatch_roll_trace.jsonl"
End Function

' 設定シート B3 以降にデバッグ用依頼NOが1件以上あるか（最大500行・連続30空で打切り。planning_core と同趣旨）
Private Function 設定シート_B列デバッグ依頼NO指定あり() As Boolean
    Dim ws As Worksheet
    Dim r As Long
    Dim nBlank As Long
    Dim v As Variant
    On Error Resume Next
    Set ws = ThisWorkbook.Worksheets("設定")
    On Error GoTo 0
    If ws Is Nothing Then Exit Function
    nBlank = 0
    For r = 3 To 502
        v = ws.Cells(r, 2).Value
        If IsError(v) Then
            nBlank = nBlank + 1
        ElseIf Len(Trim$(CStr(v))) > 0 Then
            設定シート_B列デバッグ依頼NO指定あり = True
            Exit Function
        Else
            nBlank = nBlank + 1
        End If
        If nBlank >= 30 Then Exit For
    Next r
End Function

' 段階1のみ（cmd なし・xlwings Show Console 推奨）。本番の「段階1_コア実行」とは別経路で、
' LOG シート取り込み等の VBA 後処理は行わない。止まるときは RunPython の呼び方と xlwings.conf.json を確認。
Sub 配台デバッグ_段階1_xlwings()
    On Error GoTo EH
    Application.StatusBar = "ブックを保存しています（Python がディスク上のブックを読みます）..."
    DoEvents
    ThisWorkbook.Save
    Application.StatusBar = False
    RunPython "import xlwings_console_runner as _xw; _xw.run_stage1_for_xlwings()"
    Exit Sub
EH:
    MsgBox "xlwings 段階1で失敗 (" & CStr(Err.Number) & "): " & Err.Description, vbCritical
End Sub

' JSONL をシート「配台_ロールトレース」へ読み込むだけ（段階2は従来どおり別マクロの場合は先に実行）
Sub 配台デバッグ_ロールトレースをシートに表示()
    On Error GoTo EH
    If Len(Dir$(DbgRollTraceJsonlPath())) = 0 Then
        MsgBox "ロールトレースJSONLがありません。まず「配台デバッグ_段階2とロールトレースシート」を実行してください。", vbInformation
        Exit Sub
    End If
    DbgRunConsoleRunner "run_refresh_dispatch_roll_trace_sheet_for_xlwings"
    Exit Sub
EH:
    MsgBox "xlwings 実行に失敗 (" & CStr(Err.Number) & "): " & Err.Description, vbCritical
End Sub

' JSONL を有効にした段階2 → 続けて同じブックにロールトレースシートを表示（ワンショット調査向け）
Sub 配台デバッグ_段階2とロールトレースシート()
    On Error GoTo EH
    If 設定シート_B列デバッグ依頼NO指定あり() Then
        MsgBox "【デバッグ限定配台】" & vbCrLf & vbCrLf & _
            "シート「設定」の B3 以降に依頼NOが入力されています。" & vbCrLf & _
            "この段階2では、指定した依頼NOに該当する行だけを配台します（全件ではありません）。" & vbCrLf & vbCrLf & _
            "通常運用に戻すときは B 列を空にしてください。", vbInformation, "段階2"
    End If
    ' openpyxl はディスク上のブックを読むため、設定シート B 列を反映させる
    Application.StatusBar = "ブックを保存しています（設定シート B 列を Python が読み込みます）..."
    DoEvents
    ThisWorkbook.Save
    Application.StatusBar = False
    DbgRunConsoleRunner "run_stage2_with_roll_trace_jsonl_then_refresh_sheet_for_xlwings"
    Exit Sub
EH:
    MsgBox "xlwings 実行に失敗 (" & CStr(Err.Number) & "): " & Err.Description, vbCritical
End Sub

' アクティブシートが「配台_ロールトレース」のとき、選択行を1つ下へ（現認用の「進む」）
' ※ シートを表示したあと、最初のデータ行（行2）を選んでから繰り返し実行すると見やすい
Sub 配台デバッグ_進む_次の行を選択()
    Dim ws As Worksheet
    Dim lastRow As Long

    Set ws = ActiveSheet
    If ws Is Nothing Then Exit Sub
    If ws.Name <> "配台_ロールトレース" Then
        MsgBox "シート「配台_ロールトレース」を表示してから実行してください。", vbExclamation
        Exit Sub
    End If

    lastRow = ws.UsedRange.Rows.Count + ws.UsedRange.Row - 1
    If lastRow < 2 Then
        MsgBox "データ行がありません。", vbInformation
        Exit Sub
    End If

    If ActiveCell.Row >= lastRow Then
        MsgBox "データの最終行です。", vbInformation
        Exit Sub
    End If

    On Error Resume Next
    ws.Cells(ActiveCell.Row + 1, ActiveCell.Column).Select
    If Err.Number <> 0 Then
        Err.Clear
        ws.Cells(2, 1).Select
    End If
    On Error GoTo 0

    ActiveWindow.ScrollRow = ActiveCell.Row
End Sub
