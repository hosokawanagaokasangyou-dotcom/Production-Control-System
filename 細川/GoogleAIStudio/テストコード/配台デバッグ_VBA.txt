' =========================================================
' 配台デバッグ: cmd.exe 経由で 1 ロールトレースを確認する（標準モジュールに貼り付け）
' =========================================================
' xlwings は使わない。python を cmd.exe から実行する。
' =========================================================

Private Function DbgWorkbookDir() As String
    DbgWorkbookDir = ThisWorkbook.Path
End Function

Private Function DbgPythonExe() As String
    DbgPythonExe = "C:\Users\0585\AppData\Local\Programs\Python\Python314\python.exe"
End Function

Private Function DbgRollTraceJsonlPath() As String
    DbgRollTraceJsonlPath = DbgWorkbookDir() & "\log\dispatch_roll_trace.jsonl"
End Function

Private Sub DbgRunCmdWait(ByVal cmdLine As String)
    Dim sh As Object
    Dim p As Object
    Set sh = CreateObject("WScript.Shell")
    Set p = sh.Exec("cmd.exe /c """ & cmdLine & """")
    Do While p.Status = 0
        DoEvents
    Loop
    If p.ExitCode <> 0 Then
        Err.Raise vbObjectError + 521, "DbgRunCmdWait", _
            "cmd 実行失敗 ExitCode=" & CStr(p.ExitCode) & vbCrLf & p.StdErr.ReadAll
    End If
End Sub

Private Sub DbgLoadRollTraceXlsxToSheet(ByVal traceXlsxPath As String)
    Dim srcWb As Workbook
    Dim srcWs As Worksheet
    Dim dstWs As Worksheet
    Dim ur As Range

    If Len(Dir$(traceXlsxPath)) = 0 Then
        Err.Raise vbObjectError + 522, "DbgLoadRollTraceXlsxToSheet", "トレースxlsxが見つかりません: " & traceXlsxPath
    End If

    On Error Resume Next
    Set dstWs = ThisWorkbook.Worksheets("配台_ロールトレース")
    On Error GoTo 0
    If dstWs Is Nothing Then
        Set dstWs = ThisWorkbook.Worksheets.Add(After:=ThisWorkbook.Worksheets(1))
        dstWs.Name = "配台_ロールトレース"
    End If

    Set srcWb = Workbooks.Open(Filename:=traceXlsxPath, ReadOnly:=True)
    Set srcWs = srcWb.Worksheets(1)
    Set ur = srcWs.UsedRange

    dstWs.Cells.Clear
    ur.Copy Destination:=dstWs.Range("A1")
    dstWs.Columns.AutoFit

    srcWb.Close SaveChanges:=False
End Sub

Private Sub DbgExportTraceXlsxByCmd()
    Dim root As String
    Dim py As String
    Dim scriptPath As String
    Dim jsonlPath As String
    Dim outXlsx As String
    Dim cmdLine As String

    root = DbgWorkbookDir()
    py = DbgPythonExe()
    scriptPath = root & "\python\planning_dispatch_debug.py"
    jsonlPath = DbgRollTraceJsonlPath()
    outXlsx = root & "\log\dispatch_roll_trace_export.xlsx"

    cmdLine = """" & py & """ """ & scriptPath & """ --jsonl """ & jsonlPath & """ --export-xlsx """ & outXlsx & """"
    DbgRunCmdWait cmdLine
    DbgLoadRollTraceXlsxToSheet outXlsx
End Sub

' ログJSONL -> 一覧シート反映（cmd 経由）
Sub 配台デバッグ_ロールトレースをシートに表示()
    On Error GoTo EH
    If Len(Dir$(DbgRollTraceJsonlPath())) = 0 Then
        MsgBox "ロールトレースJSONLがありません。まず「配台デバッグ_段階2とロールトレースシート」を実行してください。", vbInformation
        Exit Sub
    End If
    DbgExportTraceXlsxByCmd
    Exit Sub
EH:
    MsgBox "実行エラー: " & Err.Description, vbCritical
End Sub

' 段階2（cmd）実行 -> ロールトレースを一覧シートに反映
Sub 配台デバッグ_段階2とロールトレースシート()
    On Error GoTo EH

    Dim root As String
    Dim py As String
    Dim stage2Path As String
    Dim cmdLine As String

    root = DbgWorkbookDir()
    py = DbgPythonExe()
    stage2Path = root & "\python\plan_simulation_stage2.py"

    cmdLine = "set TASK_INPUT_WORKBOOK=""" & ThisWorkbook.FullName & """ && " & _
              "set DISPATCH_ROLL_TRACE_JSONL=""" & DbgRollTraceJsonlPath() & """ && " & _
              """" & py & """ """ & stage2Path & """"
    DbgRunCmdWait cmdLine

    DbgExportTraceXlsxByCmd
    Exit Sub
EH:
    MsgBox "実行エラー: " & Err.Description, vbCritical
End Sub

' アクティブシートが「配台_ロールトレース」のとき、選択行を1つ下へ（現認用の「進む」）
Sub 配台デバッグ_進む_次の行を選択()
    Dim ws As Worksheet
    Dim lastRow As Long

    Set ws = ActiveSheet
    If ws Is Nothing Then Exit Sub
    If ws.Name <> "配台_ロールトレース" Then
        MsgBox "シート「配台_ロールトレース」を表示してから実行してください。", vbExclamation
        Exit Sub
    End If

    lastRow = ws.UsedRange.Rows.Count + ws.UsedRange.Row - 1
    If lastRow < 2 Then
        MsgBox "データ行がありません。", vbInformation
        Exit Sub
    End If

    If ActiveCell.Row >= lastRow Then
        MsgBox "データの最終行です。", vbInformation
        Exit Sub
    End If

    On Error Resume Next
    ws.Cells(ActiveCell.Row + 1, ActiveCell.Column).Select
    If Err.Number <> 0 Then
        Err.Clear
        ws.Cells(2, 1).Select
    End If
    On Error GoTo 0

    ActiveWindow.ScrollRow = ActiveCell.Row
End Sub
