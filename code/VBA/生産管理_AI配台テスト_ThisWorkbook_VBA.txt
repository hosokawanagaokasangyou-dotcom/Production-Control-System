' ============================================================
' 貼り付け先: 標準モジュールではない ThisWorkbook（例: 生産管理_AI配台(RC1).xlsm）
' 標準モジュール: 生産管理_AI配台_xlsm_VBA.txt の Auto_Open / ShortcutMainSheet_* と対
' 計画実績比較ガント: 計画実績比較ガント.bas の WorkbookSheetChange 入口
'
' 補足:
' - OnKey はブックをまたぐため BeforeClose と Activate で登録解除／付け直し
' - 行ハイライト: 業務ロジックの結果_設備ガント_行ハイライト_OnSelection
' - 計画実績比較ガントの B1 ジャンプ: 計画実績比較ガント_WorkbookSheetChange入口
' ============================================================
Option Explicit

Private Sub Workbook_Open()
    On Error Resume Next
    Application.Calculation = xlCalculationAutomatic
    配台AI_共有Version照合_起動時警告
    ShortcutMainSheet_OnKeyRegister
    On Error GoTo 0
End Sub

Private Sub Workbook_Activate()
    On Error Resume Next
    ShortcutMainSheet_OnKeyRegister
    On Error GoTo 0
End Sub

Private Sub Workbook_SheetSelectionChange(ByVal sh As Object, ByVal Target As Range)
    On Error Resume Next
    Application.Run "結果_設備ガント_行ハイライト_OnSelection", sh, Target
    On Error GoTo 0
End Sub

Private Sub Workbook_SheetChange(ByVal sh As Object, ByVal Target As Range)
    On Error Resume Next
    Call 計画実績比較ガント_WorkbookSheetChange入口(sh, Target)
    On Error GoTo 0
End Sub

Private Sub Workbook_BeforeClose(Cancel As Boolean)
    On Error Resume Next
    ShortcutMainSheet_OnKeyUnregister
    On Error GoTo 0
End Sub
